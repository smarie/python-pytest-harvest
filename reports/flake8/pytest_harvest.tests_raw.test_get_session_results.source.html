254 def get_pytest_parametrize_marks(f):
255 """
256 Returns the @pytest.mark.parametrize marks associated with a function (and only those)
257  
258 :param f:
259 :return: a tuple containing all 'parametrize' marks
260 """
261 # pytest > 3.2.0
262 marks = getattr(f, 'pytestmark', None)
263 if marks is not None:
264 return tuple(m for m in marks if m.name == 'parametrize')
265 else:
266 # older versions
267 mark_info = getattr(f, 'parametrize', None)
268 if mark_info is not None:
269 # mark_info.args contains a list of (name, values)
270 if len(mark_info.args) % 2 != 0:
271 raise ValueError("internal pytest compatibility error - please report")
272 nb_parameters = len(mark_info.args) // 2
273 if nb_parameters > 1 and len(mark_info.kwargs) > 0:
274 raise ValueError("Unfortunately with this old pytest version it is not possible to have several "
275 "parametrization decorators")
276 res = []
277 for i in range(nb_parameters):
278 param_name, param_values = mark_info.args[2*i:2*(i+1)]
279 res.append(_LegacyMark(param_name, param_values, **mark_info.kwargs))
280 return tuple(res)

282 return ()
