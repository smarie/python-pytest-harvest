54 object.
55  
56 For each entry, the key is the test id, and the value is a dictionary containing:
57 - `'pytest_obj'`: the object under test, typically a test function
58 - `'pytest_status'`: the overall status (`'failing'`, `'skipped'`, `'passed'`)
59 - `'pytest_duration'`: the duration of the `'call'` step. By default this is the pytest unit (s) but if you set
60 `durations_in_ms=True` it becomes (ms)
61 - `'pytest_status_details'`: a dictionary containing step-by-step status details for all pytest steps (`'setup'`,
62 `'call'`, `'teardown'`). This is only included if `status_details=True` (not by default)
63  
64 It is possible to process the test id (the keys) using the `test_id_format` option. Let's assume that the id is
65 `pytest_steps/tests_raw/test_wrapped_in_class.py::TestX::()::test_easy[p1-p2]`. Here are the returned test ids
66 depending on the selected `test_id_format`
67 - `'function'` will return `test_easy[p1-p2]`
68 - `'class'` will return `TestX::()::test_easy[p1-p2]`
69 - `'module'` will return `test_wrapped_in_class.py::TestX::()::test_easy[...]`
70 - `'full'` will return the original id (this is the default behaviour)

72

75

77 functions) and/or module names.
78

366 def _pytest_item_matches_filter(item, filterset):
367 """Internal method used to check if item matches filter set"""
368 item_obj = item.obj
369 if item_obj in filterset:
370 return True
371 # support class methods: the item object can be a bound method while the filter is maybe not
372 elif item_obj is None:
373 # This can happen with DoctestItem
374 return False
375 elif _is_unbound_present(item_obj, filterset):
376 return True
377 elif any(item_obj.__module__ == f for f in filterset):
378 return True
379 else:
380 return False

383 def _is_unbound_present(item_obj, filterset):
384 """
385 Returns True if item_obj is a bound method and that its unbound version is in filterset
386 :param item_obj:
387 :param filterset:
388 :return:
389 """
390 if sys.version_info >= (3,):
391 # Python 3
392 not_bound_fct = getattr(item_obj, '__func__', None)
393 if not_bound_fct is None:
394 return False
395 else:
396 return not_bound_fct in filterset
397 else:
398 # Python 2 has the concept of "unbound method" and behaves a bit diferently
399 # see https://stackoverflow.com/questions/14574641/python-get-unbound-class-method
400 not_bound_fct = getattr(item_obj, 'im_func', None)
401 if not_bound_fct is None:
402 return False
403 elif not_bound_fct in filterset:
404 return True
405 else:
406 # maybe the filterset contains a "truly unbound" method ?
407 return not_bound_fct in {getattr(f, 'im_func', None) for f in filterset}

425 def get_pytest_status(item, durations_in_ms=False, current_request=None):
426 """
427 Returns a dictionary containing item's pytest status (success/skipped/failed, duration converted to ms) for
428 each pytest phase, and a tuple synthesizing the information.
429  
430 The synthesis status contains the worst status of all phases (setup/call/teardown), or 'pending' if there are less
431 than 3 phases.
432  
433 The synthesis duration is equal to the duration of the 'call' phase (not to the sum of all phases: indeed, we are
434 mostly interested in the test call itself).
435  
436 :param item: a pytest session.item
437 :param durations_in_ms: by default `pytest` measures durations in seconds so they are outputed in this unit. You
438 can turn the flag to True to output milliseconds instead.
439 :param current_request: if a non-None `request` is provided and the item is precisely the one from the request,
440 then the status will be 'pending'
441 :return: a tuple ((test_status, test_duration), status_dct)
442 """
443  
444 # the status keys that have been stored by our plugin.py module
445 status_keys = _get_pytest_status_keys(item)
446 if len(status_keys) == 0:
447 if current_request is not None and current_request.node == item:
448 # do not raise a warning: it is normal that there is no information, the node is being called.
449 test_status = 'pending'
450 else:
451 # warn("[pytest-harvest] Test items status is not available. You should maybe install pytest-harvest with "
452 # "pip. If it is already the case, you case try to force-use it by adding "
453 # "`pytest_plugins = ['harvest']` to your conftest.py. But for normal use this should not be required,"
454 # "installing with pip should be enough.")
455 test_status = 'unknown'
456  
457 test_duration = None
458 status_dct = dict()
459 else:
460 # adjust duration factor according to target unit
461 duration_factor = (1000 if durations_in_ms else 1)
462  
463 # create the status dictionary for that item
464 status_dct = OrderedDict()
465 test_status = 'passed'
466 test_duration = None
467 for k in status_keys:
468 statusreport = getattr(item, k)
469 status_dct[statusreport.when] = (statusreport.outcome, statusreport.duration * duration_factor)
470 # update global test status
471 if test_status == 'passed' \
472 or (test_status == 'skipped' and statusreport.outcome != 'passed'):
473 test_status = statusreport.outcome
474 # global test duration is the duration of the "call" step only
475 if statusreport.when == "call":
476 test_duration = statusreport.duration * duration_factor
477  
478 if len(status_keys) < 3:
479 # this is an incomplete test
480 test_status = 'pending'
481  
482 return (test_status, test_duration), status_dct
